SELECT X.SUMMARY, X.DESCRIPTION, Z.NAME
FROM ISSUES X
JOIN ISSUESLISTS Z ON X.ISSUESLIST_ID = Z.ID;

2//

SELECT X.SUMMARY, X.DESCRIPTION,  U.FIRSTNAME, U.LASTNAME
FROM ISSUES X
JOIN USERS U ON X.USER_ID_ASSIGNEDTO = U.ID;

3//

SELECT U.FIRSTNAME, U.LASTNAME,   COUNT(*) TASK_QUANTITY
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID
HAVING COUNT(*) >= 1
ORDER BY U.LASTNAME, U.FIRSTNAME;
